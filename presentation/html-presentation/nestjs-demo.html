@Injectable()
export class UserService implements IUserService {
  constructor(@Inject('USER_REPOSITORY') private userRepo: UserRepository) {}

  async createUser(dto: CreateUserDto): Promise<User> {
    const user = new User(dto.name, dto.email);
    return await this.userRepo.save(user);
  }
}

@Injectable()
export class OrderService {
  constructor(
    @Inject('PAYMENT_SERVICE') private paymentService: ClientProxy,
    @Inject('INVENTORY_SERVICE') private inventoryService: ClientProxy
  ) {}
  async processOrder(order: CreateOrderDto) : Promise<Order> {
    await this.inventoryService.emit('reserve_items', order.items);
    return this.paymentService.send('process_payment', order.payment);
}}
