Domain-Driven Design 

@Entity()
export class User {
  @PrimaryKey()
  id: UserId;
  name: UserName;
}

interface UserRepository {
  findById(id: UserId);
  save(user: User);
}

NestJS Implementation Example 

@Injectable()
export class UserService implements IUserService {
  constructor(@Inject('USER_REPOSITORY') private userRepo: UserRepository) {}

  async createUser(dto: CreateUserDto): Promise {
    const user = new User(dto.name, dto.email);
    return await this.userRepo.save(user);
  }
}
